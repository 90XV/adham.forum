SET GRID_ROWS = 5
SET GRID_COLS = 5
SET MAX_TRIES = 10

// Initialize player position
SET player_row = 1
SET player_col = 1

// Randomize goal position (cannot be at 1,1)
DO:
    SET goal_row = RANDOM_INT between 1 and GRID_ROWS
    SET goal_col = RANDOM_INT between 1 and GRID_COLS
WHILE goal_row == 1 AND goal_col == 1

SET tries = 0

WHILE tries < MAX_TRIES:
    PRINT "Current position: (" + player_row + "," + player_col + ")"
    PRINT "Enter move (UP/DOWN/LEFT/RIGHT):"
    READ move

    IF move == "UP" AND player_row > 1:
        player_row = player_row - 1
    ELSE IF move == "DOWN" AND player_row < GRID_ROWS:
        player_row = player_row + 1
    ELSE IF move == "LEFT" AND player_col > 1:
        player_col = player_col - 1
    ELSE IF move == "RIGHT" AND player_col < GRID_COLS:
        player_col = player_col + 1
    ELSE:
        PRINT "Invalid move! You can't go outside the grid."
        CONTINUE  // Don't count this as a try, repeat input

    INCREMENT tries by 1

    IF player_row == goal_row AND player_col == goal_col:
        PRINT "You win!"
        EXIT LOOP

IF player_row != goal_row OR player_col != goal_col:
    PRINT "You lose!"
    PRINT "Goal was at: (" + goal_row + "," + goal_col + ")"
